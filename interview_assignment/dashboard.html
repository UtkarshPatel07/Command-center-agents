The core Node.js logic that powers this evaluation. It passes the nested JSON objects to the OpenAI API and forces strict JSON output. 

const OPENAI_API_KEY = process.env.OPENAI_API_KEY;

async function evaluateAllocation(project, teams, employees) {
    const prompt = `
You are an expert Resource Manager AI.
Evaluate employees and teams against project requirements.
Return best employee and team with reasons.

Project: ${JSON.stringify(project)}
Teams: ${JSON.stringify(teams)}
Employees: ${JSON.stringify(employees)}

Instructions:
1. Best Employee: Choose best match for project scope.
2. Best Team: Choose team whose combined skills best cover scope.
3. Output MUST be strictly valid JSON:
{
  "project": "Site Name",
  "bestEmployee": { "name": "...", "reason": "..." },
  "bestTeam": { "name": "...", "reason": "..." }
}`;

    const response = await fetch("https://api.openai.com/v1/chat/completions", {
        method: "POST",
        headers: {
            "Content-Type": "application/json",
            "Authorization": \`Bearer \${OPENAI_API_KEY}\`
        },
        body: JSON.stringify({
            model: "gpt-4",
            messages: [
                { role: "system", content: "Output strict JSON." },
                { role: "user", content: prompt }
            ],
            temperature: 0.2
        })
    });

    const data = await response.json();
    return JSON.parse(data.choices[0].message.content);
}
